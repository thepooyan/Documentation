```bash
sudo pacman -S flatpak
flatpak remotes //list remotes
flatpak remote-delete [name] //delets remotes
flatpak remote-add [name] //add remotes
flatpak search [name] //search for packages
flatpak install flathub [packageID]
flatpak update
flatpak list
flatpak list --app //lists only applications and not the dependencies
flatpak history
```
